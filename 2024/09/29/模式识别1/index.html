模式识别1

A = [1.24 1.36 1.38 1.38 1.38 1.40 1.48 1.54 1.56 1.14 1.18 1.20 1.26 1.28 1.30 1.27 1.74 1.64 1.82 1.90 1.70 1.82 1.82 2.08 1.82 1.96 1.86 2.0 2.0 1.96];
A = reshape(A, 2, []); % 将 A 重塑为 2 行的矩阵

x1 = A(:, 1:8);
n1 = 8;
x2 = A(:, 9:15);
n2 = 7;

figure;
plot(x1(1,:), x1(2,:), ‘r+’); grid on; hold on;
plot(x2(1,:), x2(2,:), ‘b+’); grid on; hold on;
title(‘两类训练样本’);

A0 = [0; 0; 0];
rho = 1;

% 假设 FUNBatchPerceptron 函数已定义
[A, LOOP] = FUNBatchPerceptron(x1’, x2’, rho, A0);

w = A(2:3);
y0 = A(1);
w0 = A(1) / norm(w);
w = w ./ norm(w);

disp([‘归一化权向量=[‘, num2str(w’), ‘]’])
disp([‘阈值权=’, num2str(w0)])

% 分界线
a = w(1);
b = w(2);
c = w0;

fimplicit(@(x,y) ax + by + c, [1 2 1 2], ‘Color’, ‘black’, ‘LineWidth’, 4);

xlabel(‘触角长度’); ylabel(‘翅膀长度’);
title(‘蠓虫分类-批处理感知器算法’);

% 测试样本分类
b1 = [1.24 1.28; 1.8 1.84];
b2 = [1.4; 2.04];
plot(b1(1,:), b1(2,:), ‘r*’); grid on; hold on;
plot(b2(1,:), b2(2,:), ‘b*’); grid on; hold on;

legend(‘第一类样本’, ‘第二类样本’, ‘分界面’, ‘test1’, ‘test2’);

% 错误测试
Ne1 = sum(w’*b1 + w0 >= 0);
Ne2 = sum(w’*b2 + w0 < 0);
Ne = Ne1 + Ne2;

title([‘蠓虫分类：批量处理感知器算法，错分样本数=’, num2str(Ne)]);
